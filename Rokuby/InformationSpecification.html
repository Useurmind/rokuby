def Extend(valueMap, callParent=true)
  if(valueMap == nil)
    return
  end
  
  if(callParent)
    super(valueMap)
  end
  
  defines = valueMap[:Defines] || valueMap[:defs]
  if(defines)
    @Defines.concat(defines)
  end
end
